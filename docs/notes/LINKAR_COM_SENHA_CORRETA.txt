# Comando para Linkar com Senha Correta

# No PowerShell, execute (substitua SUA_SENHA_CORRETA pela senha real):

$senha = 'SUA_SENHA_CORRETA'
supabase link --project-ref crpzkppsriranmeumfqs --password $senha

# OU use aspas simples diretamente:

supabase link --project-ref crpzkppsriranmeumfqs --password 'SUA_SENHA_CORRETA'

# Após linkar, verifique:

supabase status

# Se mostrar URLs do Cloud (não 127.0.0.1), está conectado!
